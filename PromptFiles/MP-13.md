```python
"""
All LLM prompt templates for the DPDP Compliance Copilot.

Centralized here for easy tuning and consistency.
All user-provided data MUST be wrapped with wrap_user_content()
before insertion into any of these templates.

Naming convention:
- CLASSIFY_PROMPT: Data field classification
- GAP_BATCH_PROMPT: Batched gap analysis (multiple obligations per call)
- GENERATE_*_PROMPT: Document generation prompts
"""

# ============================================================
# DATA CLASSIFICATION
# ============================================================

CLASSIFY_PROMPT = """You are a data privacy classification expert specializing in India's Digital Personal Data Protection Act, 2023 (DPDP Act).

Your task: Classify each data field in the schema below into one or more privacy categories.

{schema_wrapped}

CATEGORIES:
- identifiers: name, email, phone, address, aadhaar, PAN, passport, user IDs
- financial: bank accounts, UPI, credit/debit cards, income, transactions, billing
- health: medical records, prescriptions, biometrics (fingerprint, face ID), diagnoses
- children: any data related to users under 18 (age, school, grade, parental info)
- sensitive: caste, religion, political opinion, sexual orientation, genetic data
- behavioral: browsing history, purchase history, location/GPS tracking, app usage, cookies
- communication: chat logs, call records, email content, SMS, notifications

RULES:
- A field can belong to MULTIPLE categories (e.g., "parent_email" is both identifiers and children)
- If unsure, classify conservatively (higher risk category)
- "high" risk = financial, health, children, sensitive data
- "medium" risk = identifiers, behavioral, communication
- "low" risk = non-personal data (but still classify if it could be PII)

Respond with ONLY a JSON object in this exact format, no other text:
{{
  "classifications": [
    {{
      "field_name": "exact field name from schema",
      "categories": ["category1", "category2"],
      "risk_level": "high" | "medium" | "low",
      "reasoning": "brief explanation of why this classification",
      "confidence": 0.0 to 1.0
    }}
  ]
}}"""


# ============================================================
# GAP ANALYSIS (BATCHED)
# ============================================================

GAP_BATCH_PROMPT = """You are a DPDP Act 2023 compliance auditor. Analyze whether the existing privacy policy adequately addresses each of the following DPDP obligations.

RELEVANT DPDP ACT TEXT:
{dpdp_context}

EXISTING PRIVACY POLICY:
{policy_wrapped}

OBLIGATIONS TO CHECK:
{obligations_list}

For EACH obligation listed above, evaluate whether the privacy policy addresses it.

SCORING RULES:
- "compliant" = policy fully and specifically addresses this obligation
- "partial" = policy mentions the topic but is vague, incomplete, or lacks DPDP-specific language
- "missing" = policy does not address this obligation at all
- If no privacy policy was provided (marked as "[No privacy policy provided]"), mark ALL as "missing"

Respond with ONLY a JSON array, no other text:
[
  {{
    "obligation": "obligation_category_name",
    "section_ref": "Section X, Rule Y",
    "status": "compliant" | "partial" | "missing",
    "gap_description": "specific description of what is missing or insufficient",
    "recommended_action": "concrete action the Data Fiduciary should take",
    "severity": "critical" | "high" | "medium" | "low",
    "confidence": 0.0 to 1.0,
    "matched_dpdp_text": "the specific DPDP text that this obligation comes from"
  }}
]

SEVERITY GUIDELINES:
- critical: Fundamental requirements like consent, notice, or breach notification completely missing
- high: Important requirements partially addressed or key elements missing
- medium: Secondary requirements or minor gaps
- low: Nice-to-have improvements"""


# ============================================================
# DOCUMENT GENERATION
# ============================================================

GENERATE_PRIVACY_NOTICE_PROMPT = """You are a legal document expert specializing in India's DPDP Act 2023.

Generate a professional privacy notice for the following company and data processing activities.

Company: {company_name}
Contact: {contact_email}
DPO: {dpo_name}
Grievance Officer: {grievance_email}

DATA COLLECTED:
{data_categories_text}

APPLICABLE OBLIGATIONS:
{obligations_text}

Generate a complete privacy notice that:
1. Cites specific DPDP Act sections (Section 5, 6, etc.)
2. Lists all personal data collected and purposes
3. Explains Data Principal rights (Sections 11-14)
4. Includes consent withdrawal mechanism
5. Addresses children's data if applicable (Section 9)
6. Includes grievance redressal contact
7. Is written in clear, plain language

Respond with the privacy notice as plain text (NOT JSON). Use markdown headers for sections."""


GENERATE_CONSENT_TEXT_PROMPT = """You are a legal document expert specializing in India's DPDP Act 2023.

Generate short, clear consent collection texts (for banners, modals, and forms) for:

Company: {company_name}
Data collected: {data_categories_text}
Purposes: {purposes_text}

Generate THREE versions:
1. COOKIE/DATA-USE BANNER: 2-3 sentences for a website banner
2. CONSENT MODAL: Short paragraph for a consent popup
3. FORM CHECKBOX TEXT: Single sentence for a checkbox label

Each must reference DPDP Act Section 6 requirements:
- Must be specific about what data and why
- Must mention right to withdraw consent
- Must be clear and unambiguous

Respond as plain text with headers for each version."""


GENERATE_RETENTION_PROMPT = """You are a data governance expert specializing in India's DPDP Act 2023.

Generate a data retention matrix for:

Company: {company_name}
Data categories: {data_categories_text}
Obligations: {obligations_text}

Create a retention matrix table with columns:
| Data Category | Data Fields | Retention Period | Legal Basis | Deletion Trigger |

Rules per DPDP Act Section 8(7):
- Data must be erased when purpose is fulfilled or consent is withdrawn
- Retention periods must be reasonable and justified
- Suggest specific timeframes based on industry practice

Respond as plain text with a markdown table."""


GENERATE_BREACH_SOP_PROMPT = """You are a cybersecurity compliance expert specializing in India's DPDP Act 2023.

Generate a Personal Data Breach Incident Response SOP for:

Company: {company_name}
DPO: {dpo_name}
Contact: {contact_email}

The SOP must address DPDP Act Section 8(6) and Rule 6 requirements:

1. DETECTION & IDENTIFICATION (Steps to identify a breach)
2. CONTAINMENT (Immediate actions to limit damage)
3. ASSESSMENT (Evaluate scope, data affected, risk level)
4. NOTIFICATION — DATA PROTECTION BOARD (Timeline, format, content per Rule 6)
5. NOTIFICATION — AFFECTED DATA PRINCIPALS (Timeline, content, channels)
6. REMEDIATION (Fix root cause, prevent recurrence)
7. DOCUMENTATION (Records to maintain)
8. POST-INCIDENT REVIEW (Lessons learned)

Include specific timelines from DPDP Rules 2025.
Respond as plain text with numbered sections and sub-steps."""
```
